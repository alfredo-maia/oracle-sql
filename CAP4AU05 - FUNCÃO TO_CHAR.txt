--CONVERSÃO DE DATA EM STRING
SELECT SYSDATE,To_Char(SYSDATE,'DD') FROM DUAL;
--CONVERSÃO DE DATA COMPLETA SEM HORA
SELECT SYSDATE,To_Char(SYSDATE,'DD/MM/YYYY') FROM DUAL;
--SO MES
SELECT SYSDATE,To_Char(SYSDATE,'MM') FROM DUAL;
--SO ANO
SELECT SYSDATE,To_Char(SYSDATE,'YYYY') FROM DUAL;
SELECT SYSDATE, To_Char(SYSDATE,'YY') FROM DUAL;
--MES
SELECT SYSDATE, To_Char(SYSDATE,'MONTH') FROM DUAL;
--DIA DA SEMANA
SELECT SYSDATE,To_Char(SYSDATE,'D') FROM DUAL;
--DIA DA SEMANA COM 3 PRIMEIRAS LETRAS
SELECT SYSDATE,To_Char(SYSDATE,'DY') FROM DUAL;
--DIA DA SEMANA COMPLETO
SELECT SYSDATE,To_Char(SYSDATE,'DAY') FROM DUAL;
--ANO
SELECT SYSDATE,To_Char(SYSDATE,'YEAR') FROM DUAL;
--DATA POR EXTENSO (fm REMOVE ESPAÇO)
SELECT To_Char(SYSDATE,'"NOVO HAMBURGO", fmDAY ","  DD "de" fmMonth "de" YYYY') FROM DUAL;
--HORAS E MINUTOS
SELECT To_Char(SYSDATE,'HH24:MI') HORA_MIN FROM DUAL;
--HORAS MINUTOS E SEGUNDOS
SELECT To_Char(SYSDATE,'HH24:MI:SS') HORA_SS FROM DUAL;
--DIA E MES COM HORA E MINUTO
SELECT To_Char(SYSDATE,'DD/MM HH24:MI:SS') FROM DUAL;
--DIA MES ANO E HORA MINUTO E SEGUNDO
SELECT To_Char(SYSDATE,'DD/MM/YYYY HH24:MI:SS') FROM DUAL;
--FORMATANDO VALORES  (L CIFRÃO, G VIRGULA, D PONTO)
SELECT NOME, To_Char(SALARIO,'L99999.99') FROM TALUNO;
SELECT NOME,To_Char(SALARIO,'L99G999D99') FROM TALUNO;
--REMOVENDO ESPAÇOS EM BRANCO
SELECT NOME,Trim(To_Char(SALARIO,'L99G999D99')) FROM TALUNO;
--CIFRAO COM CONCATENACAO
SELECT 'R$ - ' || Trim(To_Char(SALARIO,'99G999D99')) FROM TALUNO;
--CIFRA COM CONCATENACAO CURSO
SELECT 'R$ ' || (Round(SALARIO,2)) AS SALARIO FROM TALUNO;
