-- CRIANDO SEQUÊNCIA

CREATE SEQUENCE SEQ_ALUNO1
START WITH 60
  --valor inicial
INCREMENT BY 2
  --qtd a incrementar
  MINVALUE 60
  --valor mínimo
  MAXVALUE 100
  --valor máximo
  NOCACHE
  --não guarda em cache a faixa de valores
  NOCYCLE
  --não recomeça a contagem
  
  --INSERINDO USANDO SEQUÊNCIA
  
  INSERT INTO TALUNO (COD_ALUNO, NOME)
  VALUES (SEQ_ALUNO1.NEXTVAL, 'MT');
  
  SELECT * FROM TALUNO;
  
  --DESCOBRIR O VALOR ATUAL DA SEQUENCIA
  
  SELECT SEQ_ALUNO.CURRVAL FROM DUAL;
  
  --ALTERANDO O VALOR DE UMA SEQUENCIA
  
  ALTER SEQUENCE SEQ_ALUNO1 MAXVALUE 500;
  
  --DELETANDO UMA SEQUENCIA
  
  DROP SEQUENCE SEQ_ALUNO1;
