SELECT DATA, SYSDATE, DATA + 5 FROM TCONTRATO;

SELECT SYSDATE - DATA AS DIF_DIAS FROM TCONTRATO;

SELECT Trunc(SYSDATE - DATA) AS DIF_DIAS FROM TCONTRATO;

--SOMANDO HORAS NUMA DATA
SELECT SYSDATE, SYSDATE + 1 / 24 AS ADD_HORAS FROM TCONTRATO;
--SOMANDO MINUTOS NUMA DATA
SELECT SYSDATE, SYSDATE + 30 / 1440 AS ADD_MINUTOS FROM TCONTRATO;
--SOMANDO SEGUNDOS NUMA DATA
SELECT SYSDATE, SYSDATE + 30 / (3600 * 24) AS ADD_SEGUNDOS FROM TCONTRATO;
--SELEC BASICO
SELECT SYSDATE, Trunc(SYSDATE) FROM DUAL;
-- DIFERENCA DE MESES
SELECT Months_Between(SYSDATE,SYSDATE-90) AS DIF_MES FROM DUAL;
--ADICIONANDO MESES
SELECT SYSDATE,Add_Months(SYSDATE,5) AS ADD_MES FROM DUAL;
--PROXIMA DATA A PARTI DO DIA DA SEMANA
SELECT Next_Day(SYSDATE,'WEDNESDAY') AS PROXIMA_QUARTA FROM DUAL;
--ULTIMO DIA DO MES
SELECT Last_Day(SYSDATE + 360*2) FROM DUAL;
