SELECT DESCONTO , 
       Nvl(DESCONTO,0)  ,
       TOTAL  ,
       Nvl(DESCONTO,0) + TOTAL,
       Nvl2(DESCONTO,TOTAL,-1) --SE FOR NULO RETORNA -1 SE NÃO RETORNA TOTAL
 FROM TCONTRATO;
